function evenOrOdd(num){
  if (num % 2 == 0){
    return 'even'
  }
  return 'odd' 
}

console.log(evenOrOdd(2)) // 'even'
console.log(evenOrOdd(3)) // 'odd'
console.log(evenOrOdd(0)) // 'even'
console.log(evenOrOdd(-2)) // 'even'
console.log(evenOrOdd(-1))
